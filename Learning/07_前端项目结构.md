### 2.4 permission.js - 路由守卫

**`src/permission.js`：**

```javascript
import router from './router'
import { getToken } from '@/utils/auth'
import { ElMessage } from 'element-plus'
import useUserStore from '@/store/modules/user'

const whiteList = ['/login']

router.beforeEach(async (to, from, next) => {
  // 设置页面标题
  document.title = to.meta.title ? `${to.meta.title} - NeoCC` : 'NeoCC'
  
  // 判断是否已登录
  const hasToken = getToken()
  
  if (hasToken) {
    if (to.path === '/login') {
      next({ path: '/' })
    } else {
      // 获取用户信息
      const userStore = useUserStore()
      if (userStore.roles.length === 0) {
        try {
          await userStore.getInfo()
          next()
        } catch (error) {
          ElMessage.error('获取用户信息失败')
          next('/login')
        }
      } else {
        next()
      }
    }
  } else {
    if (whiteList.includes(to.path)) {
      next()
    } else {
      next(`/login?redirect=${to.fullPath}`)
    }
  }
})
```

### 2.5 store/modules/user.js - 用户状态

**`src/store/modules/user.js`：**

```javascript
import { defineStore } from 'pinia'
import { getToken, setToken, removeToken } from '@/utils/auth'
import { login as loginApi, getInfo } from '@/api/login'

export const useUserStore = defineStore('user', {
  state: () => ({
    token: getToken() || '',
    id: '',
    username: '',
    roles: []
  }),
  
  getters: {
    isLoggedIn: (state) => !!state.token,
    isAdmin: (state) => state.roles.includes('admin')
  },
  
  actions: {
    async login(loginForm) {
      const res = await loginApi(loginForm)
      const token = res.data?.token
      if (token) {
        setToken(token)
        this.token = token
      }
    },
    
    async getInfo() {
      const res = await getInfo()
      const data = res.data || res
      this.id = data.userId
      this.username = data.username
      this.roles = data.roles || []
    },
    
    logout() {
      this.token = ''
      this.roles = []
      removeToken()
    }
  }
})
```

### 2.6 utils/request.js - Axios 封装

**`src/utils/request.js`：**

```javascript
import axios from 'axios'
import { ElMessage } from 'element-plus'
import { getToken } from '@/utils/auth'

const service = axios.create({
  baseURL: '',
  timeout: 10000
})

// 请求拦截器
service.interceptors.request.use(config => {
  config.baseURL = getBaseURL(config.url)
  
  const token = getToken()
  if (token) {
    config.headers['Authorization'] = 'Bearer ' + token
  }
  
  return config
})

// 响应拦截器
service.interceptors.response.use(
  response => {
    const res = response.data
    
    if (res.code && res.code !== 200) {
      ElMessage.error(res.message || '请求失败')
      return Promise.reject(new Error(res.message))
    }
    
    return res
  },
  error => {
    ElMessage.error(error.message || '网络错误')
    return Promise.reject(error)
  }
)

function getBaseURL(url) {
  if (url.startsWith('/customer') || url.startsWith('/contract')) {
    return '/sales/api'
  }
  if (url.startsWith('/loanAudit') || url.startsWith('/bank')) {
    return '/finance/api'
  }
  return '/dev-api'
}

export default service
```

  })
